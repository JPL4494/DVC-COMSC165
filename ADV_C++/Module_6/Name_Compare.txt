#include <iostream>
#include <cstring>

using namespace std;

int main()
{

    if (length1 == length2)

        for (int i = 0; i < length1; i++)

        if (name1[i] == name2[i])

                yes = true;
            }
            else
            {
                yes = false;
                break;
            }
        }
        if (yes)
        {
            cout << endl << "The names are as follow:";
            cout << endl << name1;
            cout << endl << name2;
            cout << endl << "The names are the same" << endl;
        }
        else
        {
            length1 = length1 + 9;
        }
    }

    if (length1 != length2)
    {

        if (strcmp(name1,name2) > 0)
        {

        }

        else
        {
             cout << endl << "The names are as follow:";
             cout << endl << name1;
             cout << endl << name2;
             cout << endl;
        }

    }
